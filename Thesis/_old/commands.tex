\makeatletter
\def\maxwidth#1{\ifdim\Gin@nat@width>#1 #1\else\Gin@nat@width\fi}
\makeatother

\newcommand{\singlefigure}[3]{ %
	\begin{figure}[!ht] %
		\begin{center} %
			\includegraphics[width=0.8\maxwidth{\textwidth}]{#2} %
			\caption{#3} %
			#1 %
		\end{center} %
	\end{figure} %
}

\newcommand{\sidebysidefigures}[6]{ %
	\begin{figure}[!ht] %
		\begin{center} %
			\begin{minipage}[b]{0.45\textwidth} %
				\begin{center} %
					\includegraphics[width=\maxwidth{\textwidth}]{#2} %
					\caption{#3} %
					#1 %
				\end{center} %
			\end{minipage} %
%
			\hfill %
%
			\begin{minipage}[b]{0.45\textwidth} %
				\begin{center} %
					\includegraphics[width=\maxwidth{\textwidth}]{#5} %
					\caption{#6} %
					#4 %
				\end{center} %
			\end{minipage} %
		\end{center} %
	\end{figure} %
}
